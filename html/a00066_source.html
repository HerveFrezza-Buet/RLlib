113  static void drawingPosition(int& w, int& h, phase_type p) {
114  switch(p) {
115  case start:
116  w = 0;
117  h = WIDTH;
118  break;
119  case goal:
120  w = LENGTH-1;
121  h = WIDTH;
122  break;
123  default:
124  if(p<0 || p >= LENGTH*WIDTH+2) {
125  std::ostringstream os;
126  os << "Cliff<" << LENGTH << ',' << WIDTH << ">::drawingPosition(w,h,"
127  << p << ") : Out of bounds.";
128  throw BadState(os.str());
129  }
130 
131  p--;
132  w = p%LENGTH;
133  h = WIDTH-1-p/LENGTH;
134  break;
135  }
136  }

349  void step(const action_type a) {
350  phase_type s = current_state-1; // Easier index
351  r = StepReward();
352 
353  switch(a) {
354  case actionNorth:
355  if(s / CLIFF::length < CLIFF::width-1) // not upper wall
356  s += CLIFF::length;
357  break;
358  case actionSouth:
359  if(s / CLIFF::length > 0) // not on the edge
360  s -= CLIFF::length;
361  else if(s == 0)
362  s = CLIFF::start-1; // On start again
363  else if(s == CLIFF::length-1)
364  s = CLIFF::goal - 1; // remember that we will do s++...
365  else {
366  s = CLIFF::start-1;
367  r = FallReward();
368  }
369  break;
370  case actionEast:
371  if(s % CLIFF::length < CLIFF::length-1) // Not on right wall
372  s++;
373  break;
374  case actionWest:
375  if(s % CLIFF::length != 0) // Not on left wall
376  s--;
377  break;
378  default:
379  std::ostringstream ostr;
380  ostr << "cliff_walking::Simulator::timeStep(" << a << ")";
381  throw BadAction(ostr.str());
382  }
383 
384  current_state = s+1;
385  }
