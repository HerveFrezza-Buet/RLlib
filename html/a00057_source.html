296  void step(const action_type a) {
297  phase_type s = current_state-1; // Easier index
298  r = StepReward();
299 
300  switch(a) {
301  case actionNorth:
302  if(s / CLIFF::length < CLIFF::width-1) // not upper wall
303  s += CLIFF::length;
304  break;
305  case actionSouth:
306  if(s / CLIFF::length > 0) // not on the edge
307  s -= CLIFF::length;
308  else if(s == 0)
309  s = CLIFF::start-1; // On start again
310  else if(s == CLIFF::length-1)
311  s = CLIFF::goal - 1; // remember that we will do s++...
312  else {
313  s = CLIFF::start-1;
314  r = FallReward();
315  }
316  break;
317  case actionEast:
318  if(s % CLIFF::length < CLIFF::length-1) // Not on right wall
319  s++;
320  break;
321  case actionWest:
322  if(s % CLIFF::length != 0) // Not on left wall
323  s--;
324  break;
325  default:
326  std::ostringstream ostr;
327  ostr << "cliff_walking::Simulator::timeStep(" << a << ")";
328  throw BadAction(ostr.str());
329  }
330 
331  current_state = s+1;
332  }
