155  void timeStep(const action_type& a) {
156  double aa;
157  double acc,cphi;
158 
159  switch(a) {
160  case actionRight:
161  aa = 1;
162  break;
163  case actionLeft:
164  aa = -1;
165  break;
166  case actionNone:
167  aa = 0;
168  break;
169  default:
170  std::ostringstream ostr;
171  ostr << "inverted_pendulum::Simulator::timeStep(" << a << ")";
172  throw BadAction(ostr.str());
173  }
174 
175  aa += param_type::actionNoise()*rl::random::uniform(-1,1);
176  aa *= Param::strength();
177 
178  cphi = cos(current_state.angle);
179  acc = ( Param::g()*sin(current_state.angle)
180  - .5*Param::aml()*sin(2*current_state.angle)*current_state.speed*current_state.speed
181  - Param::a()*cphi*aa )
182  / ( 4*Param::l()/3.0 - Param::aml()*cphi*cphi );
183 
184  current_state.angle += current_state.speed*Param::tau();
185  current_state.speed += acc*Param::tau();
186 
187  if(fabs(current_state.angle)>M_PI_2) {
188  r = -1;
189  throw rl::exception::Terminal("Pendulum has fallen down");
190  }
191  r = 0;
192  }
