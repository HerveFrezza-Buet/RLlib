150  /*
151  Gradient of the log of the policy
152  Here the policy is a softmax
153  */
154  void grad_actor(gsl_vector* grad, const STATE& s, const ACTION& a) {
155  gsl_vector_set_zero(grad);
156 
157  // We need to get the probabilities of the actions
158  std::vector<double> probaActions(_nb_actions);
159  double psum = 0.0;
160  auto aiter = _action_begin;
161  auto piter = probaActions.begin();
162  while(aiter != _action_end) {
163  *piter = exp(_actor.q_function(_state_to_idx(s), *aiter));
164  psum += *piter;
165  ++aiter;
166  ++piter;
167  }
168  for(auto& p: probaActions)
169  p /= psum;
170 
171  // And we can then compute the gradient of ln(Pi)
172  piter = probaActions.begin();
173  aiter = _action_begin;
174  while(aiter != _action_end) {
175  gsl_vector_set(grad, _nb_features * std::distance(_action_begin, aiter) + _state_to_idx(s), ((*aiter)==a) - (*piter));
176  ++aiter;
177  ++piter;
178  }
179  }

185  std::map<action_type, double> get_action_probabilities(const state_type& s) const {
186 
187  std::map<action_type, double> proba;
188 
189  // We need to get the probabilities of the actions
190  double psum = 0.0;
191  auto aiter = _action_begin;
192  while(aiter != _action_end) {
193  proba[*aiter] = exp(_actor.q_function(_state_to_idx(s), *aiter));
194  psum += proba[*aiter];
195  ++aiter;
196  }
197  for(auto& ap: proba)
198  ap.second /= psum;
199 
200 
201 
202  return proba;
203  }
