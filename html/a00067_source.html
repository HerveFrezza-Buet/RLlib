87  Simulator(void) {
88  ns = GARNET_PARAM::num_states();
89  na = GARNET_PARAM::num_actions();
90  nb = GARNET_PARAM::branching();
91 
92  current_phase = ns * (std::rand() / (RAND_MAX-1.));
93 
94  rewards = new reward_type[ns];
95  memset(rewards, 0, ns*sizeof(double));
96 
97  transition_probabilities = new std::list< std::pair<unsigned int, double> >[ns*na];
98 
99  std::vector<phase_type> next_states;
100  next_states.clear();
101  for(unsigned int i = 0 ; i < ns ; ++i)
102  next_states.push_back(i);
103  double* trans_prob = new double[nb];
104 
105  // The insertion in the transition probabilites will be sorted
106  // we create the comparison function
107  auto compare_elements = [](const std::pair<unsigned int, double>& first,
108  const std::pair<unsigned int, double>& second) -> bool {
109  return first.first < second.first;
110  };
111 
112  for(unsigned int s = 0 ; s < ns ; ++s) {
113  for(unsigned int a = 0 ; a < na ; ++a) {
114  // Shuffle the arrival states (we will take the first nb elements)
115  std::random_shuffle(next_states.begin(), next_states.end());
116  // Generate the transition probabilities
117  double sum = 0.0;
118  for(unsigned int k = 0 ; k < nb ; ++k) {
119  trans_prob[k] = std::rand()/(RAND_MAX-1.);
120  sum += trans_prob[k];
121  }
122  for(unsigned int k = 0 ; k < nb ; ++k)
123  trans_prob[k] /= sum;
124 
125  // We now fill in the transition probabilities
126  // the elements are sorted by increasing arrival state number
127  for(unsigned int k = 0 ; k < nb ; ++k)
128  transition_probabilities[s*na + a].push_back(std::make_pair(next_states[k], trans_prob[k]));
129  transition_probabilities[s*na + a].sort(compare_elements);
130  }
131  }
132 
133  // Generation of the reward
134  for(unsigned int s = 0 ; s < ns ; ++s)
135  rewards[s] = std::rand()/(RAND_MAX-1.);
136  }

147  void timeStep (const action_type &a) {
148  if(a < 0 || a >= na) {
149  std::ostringstream ostr;
150  ostr.str("");
151  ostr << "Action " << a << " not in [0; " << (na-1) << "]";
152  throw BadAction(ostr.str());
153  }
154 
155  auto transition_proba = transition_probabilities[current_phase*na + a];
156  // We have a list of arrival states with their probabilities which sum to 1
157  double p = std::rand() / (RAND_MAX-1.);
158  auto piter = transition_proba.begin();
159  auto piter_end = transition_proba.end();
160  double sum = 0.0;
161  for(; piter != piter_end; ++piter) {
162  sum += piter->second;
163  if(p <= sum)
164  break;
165  }
166  current_phase = piter->first;
167  }
168 
169  reward_type reward (void) const {
170  return rewards[current_phase];
171  }
