131  void centralDifferencesTransform(void){
132 
133  // allocate size L
134  unsigned int i;
135  gsl_vector_view this_sigmaPoint;
136  gsl_vector_view that_sigmaPoint;
137 
138  // Then the sigma points can be computed
139  // first i=0;
140  this_sigmaPoint = gsl_matrix_column(sigmaPointsSet,0);
141  gsl_vector_memcpy(&(this_sigmaPoint.vector), theta);
142  // the 1<= i <= L
143  for(i=1;i<theta_size+1; ++i){
144  this_sigmaPoint = gsl_matrix_column(sigmaPointsSet,i);
145  gsl_vector_memcpy(&(this_sigmaPoint.vector), theta);
146  that_sigmaPoint = gsl_matrix_column(sigmaTheta,i-1);
147  gsl_blas_daxpy(sqrt(theta_size+lambdaUt), &(that_sigmaPoint.vector), &(this_sigmaPoint.vector) );
148  }
149  // the L+1<= i <= 2*L
150  for(i=theta_size+1;i<theta_bound; ++i){
151  this_sigmaPoint = gsl_matrix_column(sigmaPointsSet,i);
152  gsl_vector_memcpy(&(this_sigmaPoint.vector), theta);
153  that_sigmaPoint = gsl_matrix_column(sigmaTheta,i-1-theta_size);
154  gsl_blas_daxpy(-sqrt(theta_size+lambdaUt) , &(that_sigmaPoint.vector), &(this_sigmaPoint.vector) );
155  }

158  void choleskyUpdate(double alpha, gsl_vector *x){
159 
160  /*
161  This function performs a cholesky update of the cholesky factorization sigmaTheta, that is it replaces
162  the Cholesky factorization sigmaTheta by the cholesky factorization of
163  sigmaTheta*sigmaTheta^T - alpha * x * x^T
164 
165  The algorithm is an adaptation of a LU factorization rank one update. Reference is :
166  Peter Strange, Andreas Griewank and Matthias Bollh√∂fer.
167  On the Efficient Update of Rectangular LU Factorizations subject to Low Rank Modifications.
168  Electronic Transactions on Numerical Analysis, 26:161-177, 2007.
169  alg. is given in left part of fig.2.1.
170 
171  Perhaps a more efficient algorithm exists, however it should do the work for now. And the code is probably not optimal...
172 
173  WARNING
174  */
175 
176  unsigned int i,j;
177  double tmp;
178 
179 
180 
181  // A first thing is to set SS' (chol factor) in a LU form, L being unitriangular
182  // Compute U = L^T and D = diag(L)
183  gsl_matrix_set_zero(U);
184  for(i=0; i<theta_size; ++i){
185  gsl_vector_set(D,i,gsl_matrix_get(sigmaTheta,i,i));
186  for(j=0; j<=i; ++j){
187  gsl_matrix_set(U,j,i,gsl_matrix_get(sigmaTheta,i,j));
188  }
189  }
190  // Replace L by L*D^{-1} and U by D*U
191  for(i=0; i<theta_size; ++i){
192  for(j=0; j<=i; ++j){
193  tmp = gsl_matrix_get(sigmaTheta,i,j);
194  tmp /= gsl_vector_get(D,j);
195  gsl_matrix_set(sigmaTheta,i,j,tmp);
196  tmp = gsl_matrix_get(U,j,i);
197  tmp *= gsl_vector_get(D,j);
198  gsl_matrix_set(U,j,i,tmp);
199  }
200  }
201 
202  // compute the y = alpha x vector
203  gsl_vector_memcpy(y,x);
204  gsl_vector_scale(y,alpha);
205 
206  // perform the rank 1 LU modification
207  for(i=0; i<theta_size; ++i){
208 
209  // diagonal update
210  tmp = gsl_matrix_get(U,i,i) + gsl_vector_get(x,i)*gsl_vector_get(y,i);
211  gsl_matrix_set(U,i,i,tmp);
212  tmp = gsl_vector_get(y,i);
213  tmp /= gsl_matrix_get(U,i,i);
214  gsl_vector_set(y,i,tmp);
215 
216  for(j=i+1; j<theta_size; ++j){
217  // L (that is sigmaTheta) update
218  tmp = gsl_vector_get(x,j) - gsl_vector_get(x,i)*gsl_matrix_get(sigmaTheta,j,i);
219  gsl_vector_set(x,j,tmp);
220  tmp = gsl_matrix_get(sigmaTheta,j,i) + gsl_vector_get(y,i) * gsl_vector_get(x,j);
221  gsl_matrix_set(sigmaTheta,j,i,tmp);
222  }
223 
224  for(j=i+1; j<theta_size; ++j){
225  // U update
226  tmp = gsl_matrix_get(U,i,j) + gsl_vector_get(x,i)*gsl_vector_get(y,j);
227  gsl_matrix_set(U,i,j,tmp);
228  tmp = gsl_vector_get(y,j) - gsl_vector_get(y,i) * gsl_matrix_get(U,i,j);
229  gsl_vector_set(y,j,tmp);
230  }
231  }
232 
233  // Now we want the chol decomposition
234  // first D = sqrt(diag(U));
235  for(i=0; i<theta_size; ++i){
236  tmp = gsl_matrix_get(U,i,i);
237  if(tmp<=0)
238  throw exception::NotPositiveDefiniteMatrix("in ..::KTD::choleskyUpdate");
239  gsl_vector_set(D,i,sqrt(tmp));
240  }
241  // then L = L*D;
242  for(i=0; i<theta_size; ++i){
243  for(j=0; j<theta_size; ++j){
244  tmp = gsl_matrix_get(sigmaTheta,i,j) * gsl_vector_get(D,j);
245  gsl_matrix_set(sigmaTheta,i,j,tmp);
246  }
247  }
248  // that's all folks !
249  }

252  void kalmanUpdate(const STATE& state, const ACTION& action,
253  double reward,
254  const STATE& next_state,const ACTION& next_action,
255  bool is_terminal) {
256 
257  // initializations
258  unsigned int i;
259  double d, P_r, pred_r;
260  double qValue, vValue;
261  gsl_vector sigmaPoint; /* Not a pointer ! */
262 
263  /*
264  -------Prediction Step ----------------------------------------------------
265  */
266 
267  // nothing to do for thetaPred
268  gsl_matrix_scale(sigmaTheta,sqrt(1+eta_noise));
269 
270  /*
271  -------Compute the sigma-Points and their images ---------------------------
272  */
273 
274  // compute the sigma-points (weights are initialized at the creation of the agent)
275  centralDifferencesTransform();
276 
277  //compute their images
278  if(is_terminal)
279  for(i=0; i<theta_bound; ++i){
280  sigmaPoint = gsl_matrix_column(sigmaPointsSet,i).vector;
281  qValue = q(&sigmaPoint,
282  state,action);
283  gsl_vector_set(ktdQ_images_SP,i,(double)(qValue));
284  }
285  else {
286  for(i=0; i<theta_bound; ++i){
287  sigmaPoint = gsl_matrix_column(sigmaPointsSet,i).vector;
288  qValue = q(&sigmaPoint,state,action);
289  gsl_vector_set(ktdQ_images_SP,i,
290  (double)(qValue - gamma * nextValue(next_state,next_action,i)));
291  }
292  }
293 
294 
295  /*
296  -------Compute Statistics of interest --------------------------------------
297  */
298 
299  // predicted reward
300  pred_r = w_m0 * gsl_vector_get(ktdQ_images_SP,0) ;
301  for(i=1;i<theta_bound; ++i){
302  pred_r += w_i * gsl_vector_get(ktdQ_images_SP,i);
303  }
304 
305 
306  // associated variance(s)
307  d = gsl_vector_get(ktdQ_images_SP,0) - pred_r ;
308  P_r = w_c0 * d * d ;
309  for(i=1; i<theta_bound; ++i){
310  // reward
311  d = gsl_vector_get(ktdQ_images_SP,i) - pred_r ;
312  P_r += w_i * d * d ;
313  }
314  P_r += observation_noise;
315 
316  // Correlation between parameters and reward
317  gsl_vector_set_zero(P_theta_r) ;
318  for(i=1; i<theta_bound; ++i){
319  sigmaPoint = gsl_matrix_column(sigmaPointsSet,i).vector;
320  gsl_vector_memcpy(centeredSP, &sigmaPoint) ;
321  gsl_vector_sub(centeredSP,theta) ;
322  gsl_blas_daxpy(w_i * (gsl_vector_get(ktdQ_images_SP,i) - pred_r), centeredSP, P_theta_r) ;
323  }
324 
325  /*
326  -------Correction equations --------------------------------------
327  */
328 
329  // kalman gain
330  gsl_vector_memcpy(kalmanGain, P_theta_r);
331  gsl_vector_scale(kalmanGain, 1.0/P_r);
332 
333  // Update mean
334  gsl_blas_daxpy(reward - pred_r, kalmanGain, theta);
335 
336  // Update Covariance
337  choleskyUpdate(-P_r,kalmanGain);
338  }
