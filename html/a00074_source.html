169  void timeStep(const action_type& a) {
170  double aa;
171  double acc,cphi;
172 
173  switch(a) {
174  case Action::actionRight:
175  aa = 1;
176  break;
177  case Action::actionLeft:
178  aa = -1;
179  break;
180  case Action::actionNone:
181  aa = 0;
182  break;
183  default:
184  std::ostringstream ostr;
185  ostr << "inverted_pendulum::Simulator::timeStep(" << static_cast<int>(a) << ")";
186  throw BadAction(ostr.str());
187  }
188  std::uniform_real_distribution<> dis(-param_type::actionNoise(), param_type::actionNoise());
189  aa += dis(gen);
190  aa *= Param::strength();
191 
192  cphi = cos(current_state.angle);
193  acc = ( Param::g()*sin(current_state.angle)
194  - .5*Param::aml()*sin(2*current_state.angle)*current_state.speed*current_state.speed
195  - Param::a()*cphi*aa )
196  / ( 4*Param::l()/3.0 - Param::aml()*cphi*cphi );
197 
198  current_state.angle += current_state.speed*Param::tau();
199  current_state.speed += acc*Param::tau();
200 
201  if(fabs(current_state.angle)>M_PI_2) {
202  r = -1;
203  throw rl::exception::Terminal("Pendulum has fallen down");
204  }
205  r = 0;
206  }
207 
208  reward_type reward(void) const {
209  return r;
210  }
