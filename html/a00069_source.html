170  void timeStep(const action_type& a) {
171  double aa;
172 
173  switch(a) {
174  case actionForward:
175  aa = 1;
176  break;
177  case actionBackward:
178  aa = -1;
179  break;
180  case actionNone:
181  aa = 0;
182  break;
183  default:
184  std::ostringstream ostr;
185  ostr << "mountain_car::Simulator::timeStep(" << a << ")";
186  throw BadAction(ostr.str());
187  }
188 
189  current_state.speed += (0.001*aa - 0.0025*cos(3*current_state.position));
190  current_state.saturateSpeed();
191  current_state.position += current_state.speed;
192 
193  r=0;
194  if(current_state.position < param_type::minPosition()) {
195  current_state.position = param_type::minPosition();
196  current_state.speed = 0;
197  }
198  else if(current_state.position > param_type::maxPosition()) {
199 
200  if((current_state.speed >= param_type::goalSpeed())
201  &&
202  (current_state.speed <= param_type::goalSpeed() + param_type::goalSpeedMargin())) {
203  r = 1;
204  throw rl::exception::Terminal("Goal reached");
205  }
206 
207  throw rl::exception::Terminal("Upper position bound violated");
208  }
209 
210  }

244 
245  double coef_p,coef_s;
246  double position,speed;
247  int p,s;
248  Action a;
249  typename SIMULATOR::phase_type current;
250  coef_p = (SIMULATOR::param_type::maxPosition()-SIMULATOR::param_type::minPosition())/((double)(points_per_side-1));
251  coef_s = (SIMULATOR::param_type::maxSpeed()-SIMULATOR::param_type::minSpeed())/((double)(points_per_side-1));
252  for(s=0;s<points_per_side;++s) {
253  speed = SIMULATOR::param_type::minSpeed() + coef_s*s;
254  for(p=0;p<points_per_side;++p) {
255  position = SIMULATOR::param_type::minPosition() + coef_p*p;
256  current = typename SIMULATOR::phase_type(position,speed);
257  a = policy(current);
258  if(draw_q)
259  file << position << ' ' << speed << ' ' << q(current,a) << ' ' << a << std::endl;
260  else
261  file << position << ' ' << speed << ' ' << a << std::endl;
262  }
263  file << std::endl;
264  }
